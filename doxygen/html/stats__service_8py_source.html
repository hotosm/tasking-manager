76 @staticmethod
77 async def _update_tasks_stats(
78 project: dict,
79 user: dict,
80 last_state: TaskStatus,
81 new_state: TaskStatus,
82 db: Database,
83 action="change",
84 ):
85 if new_state == last_state:
86 return project, user
87
88 # Increment counters for the new state
89 if new_state == TaskStatus.MAPPED:
90 project.tasks_mapped += 1
91
92 elif new_state == TaskStatus.VALIDATED:
93 project.tasks_validated += 1
94 elif new_state == TaskStatus.BADIMAGERY:
95 project.tasks_bad_imagery += 1
96
97 # Increment user stats if action is "change"
98 if action == "change":
99 if new_state == TaskStatus.MAPPED:
100 user.tasks_mapped += 1
101 elif new_state == TaskStatus.VALIDATED:
102 user.tasks_validated += 1
103 elif new_state == TaskStatus.INVALIDATED:
104 user.tasks_invalidated += 1
105
106 # Decrement counters for the old state
107 if last_state == TaskStatus.MAPPED:
108 project.tasks_mapped -= 1
109 elif last_state == TaskStatus.VALIDATED:
110 project.tasks_validated -= 1
111
112 elif last_state == TaskStatus.BADIMAGERY:
113 project.tasks_bad_imagery -= 1
114
115 # Undo user stats if action is "undo"
116 if action == "undo":
117 if last_state == TaskStatus.MAPPED:
118 user.tasks_mapped -= 1
119 elif last_state == TaskStatus.VALIDATED:
120 user.tasks_validated -= 1
121 elif last_state == TaskStatus.INVALIDATED:
122 user.tasks_invalidated -= 1
123
124 # Update the project and user records in the database
125 await db.execute(
126 """
127 UPDATE projects
128 SET tasks_mapped = :tasks_mapped,
129 tasks_validated = :tasks_validated,
130 tasks_bad_imagery = :tasks_bad_imagery
131 WHERE id = :project_id
132 """,
133 values={
134 "tasks_mapped": project.tasks_mapped,
135 "tasks_validated": project.tasks_validated,
136 "tasks_bad_imagery": project.tasks_bad_imagery,
137 "project_id": project.id,

697 @staticmethod
698 @cached(cache=Cache.MEMORY, key_builder=cache_key_builder, ttl=3600)
699 async def get_task_stats(
700 db: Database,
701 start_date,
702 end_date,
703 org_id=None,
704 org_name=None,
705 campaign=None,
706 project_id=None,
707 country=None,
708 ):
709 """Creates task stats for a period using the TaskStatsDTO"""
710 # Base query components
711 base_query = """
712 WITH filtered_projects AS (
713 SELECT id FROM projects
714 WHERE 1 = 1
715 {filters}
716 ),
717 aggregated_stats AS (
718 SELECT
719 DATE(action_date) AS day,
720 action_text,
721 COUNT(*) AS count
722 FROM task_history
723 WHERE action = 'STATE_CHANGE'
724 AND action_text IN ('MAPPED', 'VALIDATED', 'BADIMAGERY')
725 AND project_id IN (SELECT id FROM filtered_projects)
726 GROUP BY DATE(action_date), action_text
727 ),
728 date_series AS (
729 SELECT generate_series(
730 CAST(:start_date AS DATE),
731 CAST(:end_date AS DATE),
732 INTERVAL '1 day'
733 )::DATE AS date
734 )
735 SELECT
736 TO_CHAR(ds.date, 'YYYY-MM-DD') AS date, -- Cast date to string
737 COALESCE(SUM(CASE WHEN ag.action_text = 'MAPPED' THEN ag.count END), 0) AS mapped,
738 COALESCE(SUM(CASE WHEN ag.action_text = 'VALIDATED' THEN ag.count END), 0) AS validated,
739 COALESCE(SUM(CASE WHEN ag.action_text = 'BADIMAGERY' THEN ag.count END), 0) AS bad_imagery
740 FROM date_series ds
741 LEFT JOIN aggregated_stats ag ON ds.date = ag.day
742 GROUP BY ds.date
743 HAVING
