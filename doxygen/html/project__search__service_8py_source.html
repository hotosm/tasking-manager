432 async def _filter_projects(
433 search_dto: ProjectSearchDTO, user, db: Database, as_csv: bool = False
434 ):
435 base_query, params = await ProjectSearchService.create_search_query(
436 db, user, as_csv
437 )
438 # Initialize filter list and parameters dictionary
439 filters = []
440 if search_dto.preferred_locale or search_dto.text_search:
441 subquery_filters = []
442 if search_dto.preferred_locale:
443 subquery_filters.append("locale IN (:preferred_locale, 'en')")
444 params["preferred_locale"] = search_dto.preferred_locale
445
446 if search_dto.text_search:
447 search_text = "".join(
448 char
449 for char in search_dto.text_search
450 if char not in "@|&!><\\():'"
451 )
452 tsquery_search = " & ".join([x for x in search_text.split(" ") if x])
453 ilike_search = f"%{search_text}%"
454
455 subquery_filters.append(
456 """
457 text_searchable @@ to_tsquery('english', :tsquery_search)
458 OR name ILIKE :text_search
459 """
460 )
461 params["tsquery_search"] = tsquery_search
462 params["text_search"] = ilike_search
463
464 filters.append(
465 """
466 p.id = ANY(
467 SELECT project_id
468 FROM project_info
469 WHERE {}
470 )
471 """.format(
472 " AND ".join(subquery_filters)
473 )
474 )
475
476 if search_dto.project_statuses:
477 statuses = [
478 ProjectStatus[status].value for status in search_dto.project_statuses
479 ]
480 if not user:
481 # Remove DRAFT for non-logged-in users
482 statuses_without_draft = [
483 s for s in statuses if s != ProjectStatus.DRAFT.value
484 ]
485 if statuses_without_draft:
486 filters.append("p.status = ANY(:statuses_without_draft)")
487 params["statuses_without_draft"] = tuple(statuses_without_draft)
488 else:
489 # If only DRAFT was specified, show no projects
490 filters.append("FALSE")
491 elif user.role == UserRole.ADMIN.value:
492 # Admins see all projects with specified statuses
493 filters.append("p.status = ANY(:statuses)")
494 params["statuses"] = tuple(statuses)
495 else:
496 draft_status = ProjectStatus.DRAFT.value
497 if draft_status in statuses:
498 managed_projects = await ProjectSearchService.get_managed_projects(
499 user.id, db
500 )
501 if managed_projects:
502 # Allow all statuses, but restrict DRAFT to managed projects only.
503 filters.append(
504 (
505 "p.status = ANY(:statuses) "
506 "AND (p.status != :draft_status "
507 "OR p.id = ANY(:managed_projects) "
508 "OR p.author_id = :author_id)"
509 )
510 )
511 params["statuses"] = tuple(statuses)
512 params["draft_status"] = draft_status
513 params["managed_projects"] = managed_projects
514 params["author_id"] = user.id
515 else:
516 # If no managed projects, exclude DRAFT
517 filters.append(
518 "p.status = ANY(:statuses) AND p.status != :draft_status"
519 )
520 params["statuses"] = tuple(statuses)
521 params["draft_status"] = draft_status
522 else:
523 # If DRAFT not in statuses, apply filter normally
524 filters.append("p.status = ANY(:statuses)")
525 params["statuses"] = tuple(statuses)

534 filters.append(
