62 async def _create_square(x, y, zoom, db) -> geojson.MultiPolygon:
63 """
64 Refactored function to create a geojson.MultiPolygon square using raw SQL with encode databases.
65 """
66 MAXRESOLUTION = 156543.0339
67 max = MAXRESOLUTION * 256 / 2
68 step = max / (2 ** (zoom - 1))
69
70 xmin = x * step - max
71 ymin = y * step - max
72 xmax = (x + 1) * step - max
73 ymax = (y + 1) * step - max
74
75 # Create the MultiPolygon object
76 multipolygon = MultiPolygon(
77 [Polygon([(xmin, ymin), (xmax, ymin), (xmax, ymax), (xmin, ymax)])]
78 )
79
80 # Convert MultiPolygon to WKT
81 multipolygon_wkt = multipolygon.wkt
82
83 # Query to transform and get the GeoJSON
84 create_square_query = """
85 SELECT ST_AsGeoJSON(
86 ST_Transform(
87 ST_SetSRID(
88 ST_Multi(ST_GeomFromText(:multipolygon_geometry)),
89 3857
90 ),
91 4326
92 )
93 ) AS geojson
94 """
95 # Use the WKT version of the multipolygon in the SQL query
96 square_geojson_str = await db.fetch_val(
97 create_square_query, values={"multipolygon_geometry": multipolygon_wkt}
98 )
99
100 # Convert the result back to GeoJSON and return
101 return geojson.loads(square_geojson_str)

103 @staticmethod
104 async def _create_split_tasks_from_geometry(task, db) -> list:
105 """
106 Splits a task into 4 smaller tasks based on its geometry (not OSM tile).
107 """
108 task_query = """
109 SELECT ST_AsGeoJSON(geometry) AS geometry
110 FROM tasks
111 WHERE id = :task_id AND project_id = :project_id
112 """
113 task_geojson_str = await db.fetch_val(
114 task_query, values={"task_id": task.id, "project_id": task.project_id}
115 )
116 task_geojson = geojson.loads(task_geojson_str)
117 geometry = shapely_shape(task_geojson)
118 centroid = geometry.centroid
119 minx, miny, maxx, maxy = geometry.bounds
120
121 # split geometry in half vertically, then horizontally
122 split_geometries = []
123 vertical_dividing_line = LineString([(centroid.x, miny), (centroid.x, maxy)])
124 horizontal_dividing_line = LineString([(minx, centroid.y), (maxx, centroid.y)])
125
126 vertical_halves = SplitService._as_halves(
127 split(geometry, vertical_dividing_line), centroid, "x"
128 )
129 for half in vertical_halves:
130 split_geometries += SplitService._as_halves(
131 split(half, horizontal_dividing_line), centroid, "y"
132 )
133
134 # convert split geometries into GeoJSON features expected by Task
135 split_features = []
136 for split_geometry in split_geometries:
137 multipolygon_geometry_wkt = split_geometry.wkt
138 multipolygon_as_geojson_query = """
139 SELECT ST_AsGeoJSON(
140 ST_Transform(
141 ST_SetSRID(
142 ST_Multi(ST_GeomFromText(:multipolygon_geometry)),
143 4326
144 ),
145 4326
146 )
147 ) AS geojson
148 """
149 feature_geojson = await db.fetch_val(
150 multipolygon_as_geojson_query,
151 values={"multipolygon_geometry": multipolygon_geometry_wkt},
152 )
153 feature = geojson.Feature(geometry=geojson.loads(feature_geojson))
154 feature.properties["x"] = None
155 feature.properties["y"] = None
156 feature.properties["zoom"] = None
157 feature.properties["isSquare"] = False
158 split_features.append(feature)
159
160 return split_features

162 @staticmethod
163 def _as_halves(geometries, centroid, axis) -> list:
164 """
165 Divides the given geometries into two groups -- one with geometries less
166 than or equal to the centroid position (along the given axis) and one
167 with geometries greater than the centroid position -- and returns a tuple
168 of two MultiPolygons
169 """
170 first_half = [
171 g
172 for g in geometries.geoms
173 if getattr(g.centroid, axis) <= getattr(centroid, axis)
174 ]
175 second_half = [
176 g
177 for g in geometries.geoms
178 if getattr(g.centroid, axis) > getattr(centroid, axis)
179 ]
180 return (MultiPolygon(first_half), MultiPolygon(second_half))
