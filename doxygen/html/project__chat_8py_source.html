156 @staticmethod
157 async def get_messages(
158 project_id: int, db: Database, page: int, per_page: int = 20
159 ) -> ProjectChatDTO:
160 """Get all messages on the project"""
161
162 offset = (page - 1) * per_page
163 count_query = """
164 SELECT COUNT(*)
165 FROM project_chat
166 WHERE project_id = :project_id
167 """
168 messages_query = """
169 SELECT pc.id, pc.message, pc.project_id, pc.time_stamp, u.id AS user_id, u.username, u.picture_url
170 FROM project_chat pc
171 JOIN users u ON u.id = pc.user_id
172 WHERE pc.project_id = :project_id
173 ORDER BY pc.time_stamp DESC
174 LIMIT :limit OFFSET :offset
175 """
176
177 total_count = await db.fetch_val(count_query, {"project_id": project_id})
178
179 if total_count == 0:
180 return ProjectChatDTO() # Return empty DTO if no messages
181
182 messages = await db.fetch_all(
183 messages_query,
184 {"project_id": project_id, "limit": per_page, "offset": offset},
185 )
186
187 dto = ProjectChatDTO()
188
189 for message in messages:
190 chat_dto = ListChatMessageDTO(
191 id=message["id"],
192 message=message["message"],
193 picture_url=message["picture_url"],
194 timestamp=message["time_stamp"],
195 username=message["username"],
196 )
197 dto.chat.append(chat_dto)
198
199 dto.pagination = Pagination.from_total_count(
200 page=page, per_page=per_page, total=total_count
201 )
202
203 return dto
