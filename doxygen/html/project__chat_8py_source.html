97 async def get_messages(
98 project_id: int, db: Database, page: int, per_page: int = 20
99 ) -> ProjectChatDTO:
100 """Get all messages on the project"""
101
102 offset = (page - 1) * per_page
103 count_query = """
104 SELECT COUNT(*)
105 FROM project_chat
106 WHERE project_id = :project_id
107 """
108 messages_query = """
109 SELECT pc.id, pc.message, pc.project_id, pc.time_stamp, u.id AS user_id, u.username, u.picture_url
110 FROM project_chat pc
111 JOIN users u ON u.id = pc.user_id
112 WHERE pc.project_id = :project_id
113 ORDER BY pc.time_stamp DESC
114 LIMIT :limit OFFSET :offset
115 """
116
117 total_count = await db.fetch_val(count_query, {"project_id": project_id})
118
119 if total_count == 0:
120 return ProjectChatDTO() # Return empty DTO if no messages
121
122 messages = await db.fetch_all(
123 messages_query,
124 {"project_id": project_id, "limit": per_page, "offset": offset},
125 )
126
127 dto = ProjectChatDTO()
128
129 for message in messages:
130 chat_dto = ListChatMessageDTO(
131 id=message["id"],
132 message=message["message"],
133 picture_url=message["picture_url"],
134 timestamp=message["time_stamp"],
135 username=message["username"],
136 )
137 dto.chat.append(chat_dto)
138
139 dto.pagination = Pagination.from_total_count(
140 page=page, per_page=per_page, total=total_count
141 )
