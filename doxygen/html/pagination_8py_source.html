17class Pagination:
18 """Apply an offset and limit to the query based on the current page and number of
19 items per page.
20
21 Don't create pagination objects manually. They are created by
22 :meth:`.SQLAlchemy.paginate` and :meth:`.Query.paginate`.
23
24 This is a base class, a subclass must implement :meth:`_query_items` and
25 :meth:`_query_count`. Those methods will use arguments passed as ``kwargs`` to
26 perform the queries.
27
28 :param page: The current page, used to calculate the offset. Defaults to the
29 ``page`` query arg during a request, or 1 otherwise.
30 :param per_page: The maximum number of items on a page, used to calculate the
31 offset and limit. Defaults to the ``per_page`` query arg during a request,
32 or 20 otherwise.
33 :param max_per_page: The maximum allowed value for ``per_page``, to limit a
34 user-provided value. Use ``None`` for no limit. Defaults to 100.
35 :param error_out: Abort with a ``404 Not Found`` error if no items are returned
36 and ``page`` is not 1, or if ``page`` or ``per_page`` is less than 1, or if
37 either are not ints.
38 :param count: Calculate the total number of values by issuing an extra count
39 query. For very complex queries this may be inaccurate or slow, so it can be
40 disabled and set manually if necessary.
41 :param kwargs: Information about the query to paginate. Different subclasses will
42 require different arguments.
43
44 .. versionchanged:: 3.0
45 Iterating over a pagination object iterates over its items.
46
47 .. versionchanged:: 3.0
48 Creating instances manually is not a public API.
49 """

252 def iter_pages(
253 self,
254 *,
255 left_edge: int = 2,
256 left_current: int = 2,
257 right_current: int = 4,
258 right_edge: int = 2,
259 ) -> t.Iterator[int | None]:
260 """Yield page numbers for a pagination widget. Skipped pages between the edges
261 and middle are represented by a ``None``.
262
263 For example, if there are 20 pages and the current page is 7, the following
264 values are yielded.
265
266 .. code-block:: python
267
268 1, 2, None, 5, 6, 7, 8, 9, 10, 11, None, 19, 20
269
270 :param left_edge: How many pages to show from the first page.
271 :param left_current: How many pages to show left of the current page.
272 :param right_current: How many pages to show right of the current page.
273 :param right_edge: How many pages to show from the last page.
274
275 .. versionchanged:: 3.0
276 Improved efficiency of calculating what to yield.
277
278 .. versionchanged:: 3.0
279 ``right_current`` boundary is inclusive.
280
281 .. versionchanged:: 3.0
282 All parameters are keyword-only.
283 """
284 pages_end = self.pages + 1
285
286 if pages_end == 1:
287 return
288
289 left_end = min(1 + left_edge, pages_end)
290 yield from range(1, left_end)
291
292 if left_end == pages_end:
293 return
294
295 mid_start = max(left_end, self.page - left_current)
296 mid_end = min(self.page + right_current + 1, pages_end)
297
298 if mid_start - left_end > 0:
299 yield None
300
301 yield from range(mid_start, mid_end)
302
303 if mid_end == pages_end:
304 return
305
306 right_start = max(mid_end, pages_end - right_edge)
307
308 if right_start - mid_end > 0:
309 yield None
310
311 yield from range(right_start, pages_end)
