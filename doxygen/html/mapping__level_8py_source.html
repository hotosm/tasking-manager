51 async def create(data: MappingLevelCreateDTO, db: Database):
52 """Creates a new level with its requried badges assigned."""
53 async with db.transaction():
54 # compute a number that is the minimum number that is higher than
55 # all previous ordering numbers. This is so that the newly created
56 # level allways comes after all of the previous.
57 next_ordering = await db.execute(
58 "select max(ordering)+1 from mapping_levels"
59 )
60 query = """
61 INSERT INTO mapping_levels (name, approvals_required, color, ordering, is_beginner)
62 VALUES (:name, :approvals_required, :color, :ordering, :is_beginner)
63 RETURNING id;
64 """
65 level_id = await db.execute(
66 query,
67 {
68 "name": data.name,
69 "approvals_required": data.approvals_required,
70 "color": data.color,
71 "ordering": next_ordering,
72 "is_beginner": data.is_beginner,
73 },
74 )
75
76 for badge in data.required_badges:
77 query = "INSERT INTO mapping_level_badges (level_id, badge_id) VALUES (:level_id, :badge_id)"
78 await db.execute(
79 query,
80 values={
81 "level_id": level_id,
82 "badge_id": badge.id,
83 },
84 )
85
86 return await MappingLevel.get_by_id(level_id, db)
