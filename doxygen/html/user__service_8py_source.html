294
295 username_replacement = f"user_{user_id}"
296
297 try:
298 async with db.transaction():
299 # FIXME: Should we keep user_id since that will make conversations easier to follow?
300 # Keep in mind that OSM uses user_<int:user_id> on deleted accounts.
301 await db.execute(
302 query="""
303 UPDATE users
304 SET
305 city = NULL,
306 country = NULL,
307 email_address = NULL,
308 facebook_id = NULL,

317 skype_id = NULL,

319 twitter_id = NULL,

321 WHERE id = :user_id
322 """,
323 values={
324 "username": username_replacement,
325 "user_id": user_id,
326 },
327 )
328
329 await db.execute(
330 query="""
331 DELETE FROM user_licenses
332 WHERE "user" = :user_id
333 """,
334 values={"user_id": user_id},
335 )
336
337 await db.execute(
338 query="""
339 DELETE FROM user_interests
340 WHERE user_id = :user_id
341 """,
342 values={"user_id": user_id},
343 )
344
345 is_target_admin = await UserService.is_user_an_admin(user_id, db)

347 if is_target_admin:
348 query = "UPDATE users SET role = :role WHERE id = :user_id"

426 @staticmethod
427 async def get_tasks_dto(
428 user_id: int,
429 start_date: datetime.datetime = None,
430 end_date: datetime.datetime = None,
431 task_status: str = None,
432 project_status: str = None,
433 project_id: int = None,
434 page: int = 1,
435 page_size: int = 10,
436 sort_by: str = None,
437 db: Database = None,
438 ) -> UserTaskDTOs:
439 # Base query to get the latest task history actions for a user
440 base_query = (
441 select(
442 TaskHistory.project_id.label("project_id"),
443 TaskHistory.task_id.label("task_id"),
444 func.max(TaskHistory.action_date).label("max"),
445 )
446 .where(TaskHistory.user_id == user_id)
447 .group_by(TaskHistory.task_id, TaskHistory.project_id)
448 )
449
450 if task_status:
451 base_query = base_query.where(
452 TaskHistory.action_text == TaskStatus[task_status.upper()].name
453 )
454
455 if start_date:
456 base_query = base_query.where(TaskHistory.action_date >= start_date)
457
458 if end_date:
459 base_query = base_query.where(TaskHistory.action_date <= end_date)
460
461 task_id_list = base_query.alias("task_id_list")
462
463 # Query to get the number of comments per task
464 comments_query = (
465 select(
466 TaskHistory.project_id,
467 TaskHistory.task_id,
468 func.count(TaskHistory.action).label("count"),
469 )
470 .where(TaskHistory.action == "COMMENT")
471 .group_by(TaskHistory.task_id, TaskHistory.project_id)
472 ).alias("comments_query")
473
474 # Subquery for joining comments and task IDs
475 sq = (
476 select(
477 func.coalesce(comments_query.c.count, 0).label("comments"),
478 task_id_list.c.project_id,
479 task_id_list.c.task_id,
480 task_id_list.c.max,
481 )
482 .select_from(task_id_list)
483 .outerjoin(
484 comments_query,
485 and_(
486 comments_query.c.task_id == task_id_list.c.task_id,
487 comments_query.c.project_id == task_id_list.c.project_id,
488 ),
489 )
490 ).alias("sq")
491
492 # Main task query joining with subquery
493 tasks_query = (
494 select(Task, sq.c.max, sq.c.comments)
495 .select_from(Task)
496 .join(
497 sq,
498 and_(
499 Task.id == sq.c.task_id,
500 Task.project_id == sq.c.project_id,
501 ),
502 )
503 )
504
505 if sort_by == "action_date":
506 tasks_query = tasks_query.order_by(sq.c.max)
507 elif sort_by == "-action_date":
508 tasks_query = tasks_query.order_by(desc(sq.c.max))
509 elif sort_by == "project_id":
510 tasks_query = tasks_query.order_by(sq.c.project_id)
511 elif sort_by == "-project_id":
512 tasks_query = tasks_query.order_by(desc(sq.c.project_id))
513
514 if project_status:
515 tasks_query = tasks_query.where(
516 and_(
517 Task.project_id == Project.id,
518 Project.status == ProjectStatus[project_status.upper()].value,
519 )
520 )
521
522 if project_id:
523 tasks_query = tasks_query.where(Task.project_id == project_id)
524 # Pagination
525 offset = (page - 1) * page_size
526 paginated_tasks_query = tasks_query.limit(page_size).offset(offset)
527
528 # Execute the query and fetch results
529 all_tasks = await db.fetch_all(tasks_query)
530 paginated_tasks = await db.fetch_all(paginated_tasks_query)
531
532 # Create list of task DTOs from the results
533 task_list = [
534 await Task.task_as_dto(
535 row, last_updated=row["max"], comments=row["comments"], db=db
536 )
537 for row in paginated_tasks
538 ]
539
540 user_task_dtos = UserTaskDTOs()
541 user_task_dtos.user_tasks = task_list
542 user_task_dtos.pagination = Pagination.from_total_count(
543 page=int(page), per_page=int(page_size), total=len(all_tasks)
544 )
545 return user_task_dtos

549 stats_dto = UserStatsDTO()
550 user_query = """
551 SELECT id FROM users WHERE username = :username
552 """
553 user = await db.fetch_one(query=user_query, values={"username": username})
554 if not user:
555 raise NotFound(sub_code="USER_NOT_FOUND", username=username)
556 user_id = user["id"]
557 stats_query = """
558 WITH user_actions AS (
559 SELECT
560 action_text,
561 COUNT(DISTINCT (project_id, task_id)) AS action_count
562 FROM task_history
563 WHERE user_id = :user_id
564 AND action_text IN ('VALIDATED', 'INVALIDATED', 'MAPPED')
565 GROUP BY action_text
566 ),
567 others_actions AS (
568 SELECT
569 action_text,
570 COUNT(DISTINCT (project_id, task_id)) AS action_count
571 FROM task_history th
572 WHERE (project_id, task_id) IN (
573 SELECT project_id, task_id
574 FROM task_history
575 WHERE user_id = :user_id
576 AND action_text IN ('VALIDATED', 'INVALIDATED', 'MAPPED')
577 )
578 AND user_id != :user_id
579 AND action_text IN ('VALIDATED', 'INVALIDATED')
580 GROUP BY action_text
581 )
582 SELECT
583 CAST(COALESCE((SELECT SUM(action_count) FROM user_actions WHERE action_text = 'VALIDATED'), 0)
584 AS INTEGER) AS tasks_validated,
585 CAST(COALESCE((SELECT SUM(action_count) FROM user_actions WHERE action_text = 'INVALIDATED'), 0)
586 AS INTEGER) AS tasks_invalidated,
587 CAST(COALESCE((SELECT SUM(action_count) FROM user_actions WHERE action_text = 'MAPPED'), 0)
588 AS INTEGER) AS tasks_mapped,
589 CAST(COALESCE((SELECT SUM(action_count) FROM others_actions WHERE action_text = 'VALIDATED'), 0)
590 AS INTEGER) AS tasks_validated_by_others,
591 CAST(COALESCE((SELECT SUM(action_count) FROM others_actions WHERE action_text = 'INVALIDATED'), 0)
592 AS INTEGER) AS tasks_invalidated_by_others;
593 """
594 stats_result = await db.fetch_one(
595 query=stats_query, values={"user_id": user_id}
596 )

601 stats_dto.tasks_invalidated_by_others = stats_result[
602 "tasks_invalidated_by_others"
603 ]
604

606 SELECT COUNT(DISTINCT project_id) AS projects_count
607 FROM task_history
608 WHERE user_id = :user_id AND action_text = 'MAPPED';
609 """
610 projects_mapped = await db.fetch_one(

621 )

624 stats_dto.time_spent_mapping = 0
625 stats_dto.time_spent_validating = 0
626 # Total validation time
627 total_validation_time_query = """
628 WITH max_action_text_per_minute AS (
629 SELECT
630 date_trunc('minute', action_date) AS trn,
631 MAX(action_text) AS tm
632 FROM task_history
633 WHERE user_id = :user_id
634 AND action = 'LOCKED_FOR_VALIDATION'
635 GROUP BY date_trunc('minute', action_date)
636 )
637 SELECT
638 SUM(EXTRACT(EPOCH FROM to_timestamp(tm, 'HH24:MI:SS') - to_timestamp('00:00:00', 'HH24:MI:SS')))
639 AS total_time

641 """
642 result = await db.fetch_one(
643 total_validation_time_query, values={"user_id": user.id}

648
649 # Total mapping time

656 ) AS total_mapping_time_seconds

658 WHERE user_id = :user_id
659 AND action IN ('LOCKED_FOR_MAPPING', 'AUTO_UNLOCKED_FOR_MAPPING')
660 """
661 result = await db.fetch_one(
