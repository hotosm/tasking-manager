275 @staticmethod
276 async def get_tasks_dto(
277 user_id: int,
278 start_date: datetime.datetime = None,
279 end_date: datetime.datetime = None,
280 task_status: str = None,
281 project_status: str = None,
282 project_id: int = None,
283 page: int = 1,
284 page_size: int = 10,
285 sort_by: str = None,
286 db: Database = None,
287 ) -> UserTaskDTOs:
288 # Base query to get the latest task history actions for a user
289 base_query = (
290 select(
291 TaskHistory.project_id.label("project_id"),
292 TaskHistory.task_id.label("task_id"),
293 func.max(TaskHistory.action_date).label("max"),
294 )
295 .where(TaskHistory.user_id == user_id)
296 .group_by(TaskHistory.task_id, TaskHistory.project_id)
297 )
298
299 if task_status:
300 base_query = base_query.where(
301 TaskHistory.action_text == TaskStatus[task_status.upper()].name
302 )
303
304 if start_date:
305 base_query = base_query.where(TaskHistory.action_date >= start_date)
306
307 if end_date:
308 base_query = base_query.where(TaskHistory.action_date <= end_date)
309
310 task_id_list = base_query.alias("task_id_list")
311
312 # Query to get the number of comments per task
313 comments_query = (
314 select(
315 TaskHistory.project_id,
316 TaskHistory.task_id,
317 func.count(TaskHistory.action).label("count"),
318 )
319 .where(TaskHistory.action == "COMMENT")
320 .group_by(TaskHistory.task_id, TaskHistory.project_id)
321 ).alias("comments_query")
322
323 # Subquery for joining comments and task IDs
324 sq = (
325 select(
326 func.coalesce(comments_query.c.count, 0).label("comments"),
327 task_id_list.c.project_id,
328 task_id_list.c.task_id,
329 task_id_list.c.max,
330 )
331 .select_from(task_id_list)
332 .outerjoin(
333 comments_query,
334 and_(
335 comments_query.c.task_id == task_id_list.c.task_id,
336 comments_query.c.project_id == task_id_list.c.project_id,
337 ),
338 )
339 ).alias("sq")
340
341 # Main task query joining with subquery
342 tasks_query = (
343 select(Task, sq.c.max, sq.c.comments)
344 .select_from(Task)
345 .join(
346 sq,
347 and_(
348 Task.id == sq.c.task_id,
349 Task.project_id == sq.c.project_id,
350 ),
351 )
352 )
353
354 if sort_by == "action_date":
355 tasks_query = tasks_query.order_by(sq.c.max)
356 elif sort_by == "-action_date":
357 tasks_query = tasks_query.order_by(desc(sq.c.max))
358 elif sort_by == "project_id":
359 tasks_query = tasks_query.order_by(sq.c.project_id)
360 elif sort_by == "-project_id":
361 tasks_query = tasks_query.order_by(desc(sq.c.project_id))
362
363 if project_status:
364 tasks_query = tasks_query.where(
365 and_(
366 Task.project_id == Project.id,
367 Project.status == ProjectStatus[project_status.upper()].value,
368 )
369 )
370
371 if project_id:
372 tasks_query = tasks_query.where(Task.project_id == project_id)
373 # Pagination
374 offset = (page - 1) * page_size
375 paginated_tasks_query = tasks_query.limit(page_size).offset(offset)
376
377 # Execute the query and fetch results
378 all_tasks = await db.fetch_all(tasks_query)
379 paginated_tasks = await db.fetch_all(paginated_tasks_query)
380
381 # Create list of task DTOs from the results
382 task_list = [
383 await Task.task_as_dto(
384 row, last_updated=row["max"], comments=row["comments"], db=db
385 )
386 for row in paginated_tasks
387 ]
388
389 user_task_dtos = UserTaskDTOs()
390 user_task_dtos.user_tasks = task_list
391 user_task_dtos.pagination = Pagination.from_total_count(
392 page=int(page), per_page=int(page_size), total=len(all_tasks)
393 )
394 return user_task_dtos
