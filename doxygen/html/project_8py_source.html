322 def set_country_info(self):
323 """Sets the default country based on centroid"""

325 logger.debug("Skipping country lookup due to missing centroid")
326 return
327
328 centroid_wkt = None
329 if isinstance(self.centroid, str):
330 centroid_wkt = self.centroid
331
332 elif isinstance(self.centroid, bytes):
333 try:
334 centroid_geom = wkb.loads(self.centroid)

336 except Exception as e:
337 logger.debug(f"Error converting EWKB to WKT: {e}", exc_info=True)
338 return
339 else:
340 centroid_wkt = self.centroid
341
342 centroid = WKTElement(centroid_wkt, srid=4326)
343 centroid = to_shape(centroid)

354 "Referer": os.environ.get("TM_APP_BASE_URL", "https://example.com"),

1938 tasks_mapped: int,
1939 tasks_validated: int,
1940 total_tasks: int,
1941 tasks_bad_imagery: int,
1942 ) -> int:
1943 """Calculates percentages of contributions based on provided statistics."""

1947 (tasks_mapped + tasks_validated)

1961 * 100

1963 except ZeroDivisionError:
