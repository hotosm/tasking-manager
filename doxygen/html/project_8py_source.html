310 """Sets the default country based on centroid"""

312 logger.debug("Skipping country lookup due to missing centroid")
313 return
314
315 centroid_wkt = None
316 if isinstance(self.centroid, str):
317 centroid_wkt = self.centroid
318
319 elif isinstance(self.centroid, bytes):
320 try:
321 centroid_geom = wkb.loads(self.centroid)
322 centroid_wkt = centroid_geom.wkt
323 except Exception as e:
324 logger.debug(f"Error converting EWKB to WKT: {e}", exc_info=True)
325 return
326 else:
327 centroid_wkt = self.centroid
328
329 centroid = WKTElement(centroid_wkt, srid=4326)
330 centroid = to_shape(centroid)
331 lat, lng = (centroid.y, centroid.x)

333 settings.OSM_NOMINATIM_SERVER_URL, lat, lng

341 "Referer": os.environ.get("TM_APP_BASE_URL", "https://example.com"),
342 }

344 response = requests.get(url, headers=headers)
