135 SELECT DISTINCT contributor

156 """Get all contributors of a task"""
157 query = """
158 SELECT DISTINCT u.username
159 FROM task_history th
160 JOIN users u ON th.user_id = u.id
161 WHERE th.project_id = :project_id
162 AND th.task_id = :task_id
163 AND th.action != :comment_action

168 "project_id": project_id,
169 "task_id": task_id,
170 "comment_action": TaskAction.COMMENT.name,
