29 def mlb_transform(table, column, prefix):
30 """Transforms multi label column into multiple columns and retruns the data frame with new columns
31 :param table: data frame
32 :param columns: columns to transform
33 :param prefix: prefix for the new columns
34 :return: None as it modifies the data frame in place
35 """
36 mlb = MultiLabelBinarizer()
37 mlb.fit(table[column])
38 mlb_table = pd.DataFrame(mlb.transform(table[column]), columns=mlb.classes_)
39 mlb_table = mlb_table.add_prefix(prefix)
40 table = pd.concat([table, mlb_table], axis=1)
41 table = table.drop(column, axis=1)
42 return table

96 def get_similar_project_ids(all_projects_df, target_project_df):
97 """Gets top n similar projects
98 :param all_projects_df: data frame of all projects
99 :param target_project_df: data frame of target project
100 :return: list of similar project_ids

102 # Remove the target project from the all projects data frame
103 all_projects_df = all_projects_df[
104 all_projects_df["id"] != target_project_df["id"].values[0]
105 ]

113 similar_project_indices = similarity_matrix.flatten().argsort()[::-1]
114 # Get the similar project ids in the order of similarity
115 similar_projects = all_projects_df.iloc[similar_project_indices][
116 "id"
117 ].values.tolist()

132 # Query to fetch all published projects with their related data

134 SELECT p.id, p.default_locale, p.difficulty, p.mapping_types, p.country,
135 COALESCE(ARRAY_AGG(pi.interest_id), ARRAY[]::INTEGER[]) AS categories
136 FROM projects p
137 LEFT JOIN (
138 SELECT pi.project_id, i.id as interest_id
139 FROM project_interests pi
140 JOIN interests i ON pi.interest_id = i.id

142 WHERE p.status = :status
143 GROUP BY p.id
144 """
