22 def trim_grid_to_aoi(grid_dto: GridDTO) -> geojson.FeatureCollection:
23 """
24 Removes grid squares not intersecting with the aoi. Optionally leaves partially intersecting task squares
25 complete or clips them exactly to the AOI outline
26 :param grid_dto: the dto containing
27 :return: geojson.FeatureCollection trimmed task grid
28 """
29 # get items out of the dto
30 grid = geojson.loads(geojson.dumps(grid_dto.grid))
31 aoi = geojson.loads(geojson.dumps(grid_dto.area_of_interest))
32 clip_to_aoi = grid_dto.clip_to_aoi
33
34 # create a shapely shape from the aoi
35 aoi_multi_polygon_geojson = GridService.merge_to_multi_polygon(
36 aoi, dissolve=True
37 )
38
39 aoi_multi_polygon = shapely.geometry.shape(aoi_multi_polygon_geojson)
40 intersecting_features = []
41 for feature in grid["features"]:
42 # create a shapely shape for the tile
43 tile = shapely.geometry.shape(feature["geometry"])
44 if aoi_multi_polygon.contains(tile):
45 # tile is completely within aoi, use as is
46 intersecting_features.append(feature)
47 else:
48 intersection = aoi_multi_polygon.intersection(tile)
49 if intersection.is_empty or intersection.geom_type not in [
50 "Polygon",
51 "MultiPolygon",
52 ]:
53 continue # this intersections which are not polygons or which are completely outside aoi
54 # tile is partially intersecting the aoi
55 clipped_feature = GridService._update_feature(
56 clip_to_aoi, feature, intersection
57 )
58 intersecting_features.append(clipped_feature)
59 return geojson.FeatureCollection(intersecting_features)

66 :return: task features

193 def _convert_to_multipolygon(features: list) -> MultiPolygon:
194 """
195 converts a list of (multi)polygon geometries to one single multipolygon
196 :param features:
197 :return:
198 """
199 rings = []
200 for feature in features:
201 if isinstance(feature.geometry, MultiPolygon):
202 rings = rings + [geom for geom in feature.geometry.geoms]
203 else:
204 rings = rings + [feature.geometry]
205
206 geometry = MultiPolygon(rings)
207
208 # Downsample 3D -> 2D
209 if geometry.has_z:
210 geometry = shapely.ops.transform(GridService._to_2d, geometry)
211 wkt2d = geometry.wkt
212 geom2d = shapely.wkt.loads(wkt2d)
213
214 return geom2d
