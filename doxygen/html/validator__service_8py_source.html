52 """
53 Lock supplied tasks for validation
54 :raises ValidatorServiceError
55 """
56 # Loop supplied tasks to check they can all be locked for validation
57 tasks_to_lock = []
58 for task_id in validation_dto.task_ids:
59 task = await Task.get(task_id, validation_dto.project_id, db)
60
61 if task is None:
62 raise NotFound(
63 sub_code="TASK_NOT_FOUND",
64 task_id=task_id,
65 project_id=validation_dto.project_id,
66 )
67 if not (
68 task.locked_by == validation_dto.user_id
69 and TaskStatus(task.task_status) == TaskStatus.LOCKED_FOR_VALIDATION
70 ):
71 if TaskStatus(task.task_status) not in [
72 TaskStatus.MAPPED,
73 TaskStatus.INVALIDATED,
74 TaskStatus.BADIMAGERY,
75 ]:
76 raise ValidatorServiceError(
77 f"NotReadyForValidation- Task {task_id} is not MAPPED, BADIMAGERY or INVALIDATED"
78 )
79 user_can_validate = await ValidatorService._user_can_validate_task(
80 validation_dto.user_id, task.mapped_by, db
81 )
82 if not user_can_validate:
83 raise ValidatorServiceError(
84 "CannotValidateMappedTask-"
85 + "Tasks cannot be validated by the same user who marked task as mapped or badimagery"
86 )
87
88 tasks_to_lock.append(task)
89
90 (
91 user_can_validate,
92 error_reason,
93 ) = await ProjectService.is_user_permitted_to_validate(
94 validation_dto.project_id, validation_dto.user_id, db
95 )
96
97 if not user_can_validate:
98 if error_reason == ValidatingNotAllowed.USER_NOT_ACCEPTED_LICENSE:
99 raise UserLicenseError("User must accept license to map this task")
100 elif error_reason == ValidatingNotAllowed.USER_NOT_ON_ALLOWED_LIST:
101 raise ValidatorServiceError(
102 "UserNotAllowed- Validation not allowed because: User not on allowed list"
103 )
104 elif error_reason == ValidatingNotAllowed.PROJECT_NOT_PUBLISHED:
105 raise ValidatorServiceError(
106 "ProjectNotPublished- Validation not allowed because: Project not published"
107 )
108 elif error_reason == ValidatingNotAllowed.USER_ALREADY_HAS_TASK_LOCKED:
109 user_tasks = await Task.get_locked_tasks_for_user(
110 validation_dto.user_id, db
111 )
112 if set(user_tasks.locked_tasks) != set(validation_dto.task_ids):
113 raise ValidatorServiceError(
114 "UserAlreadyHasTaskLocked- User already has a task locked"
115 )
116 else:
117 raise ValidatorServiceError(
118 f"ValidtionNotAllowed- Validation not allowed because: {error_reason}"
119 )
120
121 # Lock all tasks for validation
122 dtos = []
123 for task in tasks_to_lock:
124 await Task.lock_task_for_validating(
125 task.id, validation_dto.project_id, validation_dto.user_id, db
126 )
127 dtos.append(
128 await Task.as_dto_with_instructions(
129 task.id,
130 validation_dto.project_id,
131 db,
132 validation_dto.preferred_locale,
133 )
134 )

160 async def process_task(
161 project_id,
162 task_to_unlock,
163 validated_dto,
164 message_sent_to: Set[int],
165 message_lock: asyncio.Lock,
166 ):
167 async with db_connection.database.connection() as db:
168 task = task_to_unlock["task"]
169 if task_to_unlock["comment"]:
170 await MessageService.send_message_after_comment(
171 validated_dto.user_id,
172 task_to_unlock["comment"],
173 task.id,
174 validated_dto.project_id,
175 db,
176 )
177 if (
178 task_to_unlock["new_state"] == TaskStatus.VALIDATED
179 or task_to_unlock["new_state"] == TaskStatus.INVALIDATED
180 ):
181 mapped_by = task.mapped_by
182 # acquire lock so two coros can’t both think “not yet sent”
183 async with message_lock:
184 need_to_send = mapped_by not in message_sent_to
185 if need_to_send:
186 message_sent_to.add(mapped_by)
187
188 if need_to_send:
189 await MessageService.send_message_after_validation(
190 task_to_unlock["new_state"],
191 validated_dto.user_id,
192 mapped_by,
193 task.id,
194 project_id,
195 db,
196 )
197
198 # Set last_validation_date for the mapper to current date
199 if task_to_unlock["new_state"] == TaskStatus.VALIDATED:
200 query = """
201 UPDATE users
202 SET last_validation_date = :timestamp
203 WHERE id = (
204 SELECT mapped_by
205 FROM tasks
206 WHERE id = :task_id
207 AND project_id = :project_id
208 );
209 """
210 values = {
211 "timestamp": datetime.datetime.utcnow(),
212 "task_id": task.id,
213 "project_id": validated_dto.project_id,
214 }
215 await db.execute(query=query, values=values)
216
217 # Update stats if user setting task to a different state from previous state
218 prev_status = await TaskHistory.get_last_status(project_id, task.id, db)
219 if prev_status != task_to_unlock["new_state"]:
220 await StatsService.update_stats_after_task_state_change(
221 validated_dto.project_id,
222 validated_dto.user_id,
223 prev_status,
224 task_to_unlock["new_state"],
225 db,
226 )
227 task_mapping_issues = await ValidatorService.get_task_mapping_issues(
228 task_to_unlock
229 )
230 await Task.unlock_task(
231 task_id=task.id,
232 project_id=project_id,
233 user_id=validated_dto.user_id,
234 new_state=task_to_unlock["new_state"],
235 db=db,
236 comment=task_to_unlock["comment"],
237 issues=task_mapping_issues,
238 )
239
240 return await Task.as_dto_with_instructions(
241 task.id, project_id, db, validated_dto.preferred_locale

245 """
246 Process tasks concurrently and ensure each task gets its own DB connection.
