175 @staticmethod
176 async def get_all_users(query: UserSearchQuery, db) -> UserSearchDTO:
177 """Search and filter all users"""
178
179 base_query = """
180 SELECT
181 u.id,
182 u.username,
183 u.mapping_level,
184 u.role,
185 u.picture_url,
186 us.date_obtained,
187 us.stats,
188 un.level_id::bool as requires_approval,
189 uv.level_id::bool as user_has_voted
190 FROM
191 users AS u
192 LEFT JOIN
193 user_stats AS us ON u.id = us.user_id
194 LEFT JOIN
195 user_next_level AS un ON u.id = un.user_id
196 LEFT JOIN
197 user_level_vote AS uv ON u.id = uv.user_id AND uv.voter_id = :voter_id
198 """
199 filters = []
200 params = {}
201
202 if query.mapping_level and re.match(r"^\d+$", query.mapping_level):
203 mapping_level_array = map(int, query.mapping_level.split(","))
204 filters.append("mapping_level = ANY(:mapping_levels)")
205 params["mapping_levels"] = tuple(mapping_level_array)
206
207 if query.username:
208 filters.append("username ILIKE :username")
209 params["username"] = f"%{query.username}%"
210
211 if query.role:
212 roles = query.role.split(",")
213 role_array = [UserRole[role].value for role in roles]
214 filters.append("role = ANY(:roles)")
215 params["roles"] = tuple(role_array)
216
217 if filters:
218 base_query += " WHERE " + " AND ".join(filters)
219
220 if query.sort:
221 base_query += (
222 " ORDER BY (us.stats ->> :sort)::float "
223 + query.sort_dir
224 + " NULLS LAST"
225 )
226 else:
227 base_query += " ORDER BY requires_approval, username"
228
229 if query.pagination:
230 base_query += " LIMIT :limit OFFSET :offset"
231 base_params = params.copy()
232 base_params["voter_id"] = query.voter_id
233 base_params["limit"] = query.per_page
234 base_params["offset"] = (query.page - 1) * query.per_page
235 if query.sort:
236 base_params["sort"] = query.sort
237
238 results = await db.fetch_all(base_query, base_params)
239 else:
240 results = await db.fetch_all(base_query, params)
241
242 dto = UserSearchDTO()
243 for result in results:
244 listed_user = ListedUser()
245 listed_user.id = result["id"]
246 listed_user.mapping_level = (
247 await MappingLevel.get_by_id(result["mapping_level"], db)
248 ).name
249 listed_user.username = result["username"]
250 listed_user.picture_url = result["picture_url"]
251 listed_user.stats_last_updated = result["date_obtained"]
252 listed_user.stats = json.loads(result["stats"]) if result["stats"] else None
253 listed_user.requires_approval = result["requires_approval"]
254 listed_user.user_has_voted = result["user_has_voted"]
255 listed_user.role = UserRole(result["role"]).name
256 dto.users.append(listed_user)
257
258 if query.pagination:
259 count_query = "SELECT COUNT(*) FROM users"
260 count_query += " WHERE " + " AND ".join(filters) if filters else ""
261 total_count = await db.fetch_val(count_query, params)
262 dto.pagination = Pagination.from_total_count(
263 query.page, query.per_page, total_count
264 )
265
266 return dto

278 username: str, project_id: int, page: int, db: Database
279 ) -> UserFilterDTO:
280 """Finds users that match the first characters, for auto-complete.
281
282 Users who have participated (mapped or validated) in the project, if given, will be
283 returned ahead of those who have not.
284 """
285 query = """
286 SELECT u.username, :project_id = ANY(u.projects_mapped) AS participant
287 FROM users u
288 WHERE u.username ILIKE :username || '%'
289 ORDER BY participant DESC NULLS LAST, u.username
290 LIMIT 20 OFFSET :offset
291 """
292
293 offset = (page - 1) * 20
294 values = {
295 "username": username.lower(),
296 "project_id": project_id,
297 "offset": offset,
298 }
299
300 results = await db.fetch_all(query, values=values)
301
302 if not results:
303 raise NotFound(sub_code="USER_NOT_FOUND", username=username)
304
305 dto = UserFilterDTO()
306 for result in results:
307 dto.usernames.append(result["username"])
308 if project_id is not None:
309 participant = ProjectParticipantUser(
310 username=result["username"],
311 project_id=project_id,
312 is_participant=bool(result["participant"]),
313 )
314 dto.users.append(participant)
315
316 total_query = """
317 SELECT COUNT(*) FROM users u WHERE u.username ILIKE :username || '%'
318 """
319 total = await db.fetch_val(total_query, values={"username": username.lower()})
320 dto.pagination = Pagination.from_total_count(
321 page=page, per_page=20, total=total
322 )
323
324 return dto

350 @staticmethod
351 async def get_mapped_projects(
352 user_id: int, preferred_locale: str, db: Database
353 ) -> UserMappedProjectsDTO:
354 """Get all projects a user has mapped on"""
355
356 # Subquery for validated tasks
357 query_validated = """
358 SELECT project_id, COUNT(validated_by) AS validated
359 FROM tasks
360 WHERE project_id IN (
361 SELECT unnest(projects_mapped) FROM users WHERE id = :user_id
362 ) AND validated_by = :user_id
363 GROUP BY project_id, validated_by
364 """
365
366 # Subquery for mapped tasks
367 query_mapped = """
368 SELECT project_id, COUNT(mapped_by) AS mapped
369 FROM tasks
370 WHERE project_id IN (
371 SELECT unnest(projects_mapped) FROM users WHERE id = :user_id
372 ) AND mapped_by = :user_id
373 GROUP BY project_id, mapped_by
374 """
375
376 # Union of validated and mapped tasks
377 query_union = f"""
378 SELECT COALESCE(v.project_id, m.project_id) AS project_id,
379 COALESCE(v.validated, 0) AS validated,
380 COALESCE(m.mapped, 0) AS mapped
381 FROM ({query_validated}) v
382 FULL OUTER JOIN ({query_mapped}) m
383 ON v.project_id = m.project_id
384 """
385
386 # Main query to get project details
387 query_projects = f"""
388 SELECT p.id, p.status, p.default_locale, u.mapped, u.validated, ST_AsGeoJSON(p.centroid) AS centroid
389 FROM projects p
390 JOIN ({query_union}) u ON p.id = u.project_id
391 ORDER BY p.id DESC
392 """
393
394 results = await db.fetch_all(query_projects, {"user_id": user_id})
395
396 mapped_projects_dto = UserMappedProjectsDTO()
397 for row in results:
398 mapped_project = MappedProject()
399 mapped_project.project_id = row["id"]
400 mapped_project.status = ProjectStatus(row["status"]).name
401 mapped_project.tasks_mapped = row["mapped"]
402 mapped_project.tasks_validated = row["validated"]
403 mapped_project.centroid = geojson.loads(row["centroid"])
404 project_info = await ProjectInfo.get_dto_for_locale(
405 db, row["id"], preferred_locale, row["default_locale"]
406 )
407 mapped_project.name = project_info.name
408 mapped_projects_dto.mapped_projects.append(mapped_project)
409 return mapped_projects_dto
