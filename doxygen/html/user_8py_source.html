171 """Search and filter all users"""
172
173 base_query = """
174 SELECT id, username, mapping_level, role, picture_url FROM users
175 """
176 filters = []
177 params = {}
178
179 if query.mapping_level:
180 mapping_levels = query.mapping_level.split(",")
181 mapping_level_array = [
182 MappingLevel[mapping_level].value for mapping_level in mapping_levels
183 ]
184 filters.append("mapping_level = ANY(:mapping_levels)")
185 params["mapping_levels"] = tuple(mapping_level_array)
186
187 if query.username:
188 filters.append("username ILIKE :username")
189 params["username"] = f"%{query.username}%"
190
191 if query.role:
192 roles = query.role.split(",")
193 role_array = [UserRole[role].value for role in roles]
194 filters.append("role = ANY(:roles)")
195 params["roles"] = tuple(role_array)
196
197 if filters:
198 base_query += " WHERE " + " AND ".join(filters)
199
200 base_query += " ORDER BY username"
201
202 if query.pagination:
203 base_query += " LIMIT :limit OFFSET :offset"
204 base_params = params.copy()
205 base_params["limit"] = query.per_page
206 base_params["offset"] = (query.page - 1) * query.per_page
207
208 results = await db.fetch_all(base_query, base_params)
209
210 else:
211 results = await db.fetch_all(base_query, params)
212
213 dto = UserSearchDTO()
214 for result in results:
215 listed_user = ListedUser()
216 listed_user.id = result["id"]
217 listed_user.mapping_level = MappingLevel(result["mapping_level"]).name
218 listed_user.username = result["username"]
219 listed_user.picture_url = result["picture_url"]
220 listed_user.role = UserRole(result["role"]).name
221 dto.users.append(listed_user)
222
223 if query.pagination:
224 count_query = "SELECT COUNT(*) FROM users"
225 count_query += " WHERE " + " AND ".join(filters) if filters else ""
226 total_count = await db.fetch_val(count_query, params)

251 SELECT u.username, :project_id = ANY(u.projects_mapped) AS participant
252 FROM users u
253 WHERE u.username ILIKE :username || '%'
254 ORDER BY participant DESC NULLS LAST, u.username
255 LIMIT 20 OFFSET :offset
256 """
257
258 offset = (page - 1) * 20
259 values = {
260 "username": username.lower(),
261 "project_id": project_id,

315 @staticmethod
316 async def get_mapped_projects(
317 user_id: int, preferred_locale: str, db: Database
318 ) -> UserMappedProjectsDTO:
319 """Get all projects a user has mapped on"""
320
321 # Subquery for validated tasks
322 query_validated = """
323 SELECT project_id, COUNT(validated_by) AS validated
324 FROM tasks
325 WHERE project_id IN (
326 SELECT unnest(projects_mapped) FROM users WHERE id = :user_id
327 ) AND validated_by = :user_id
328 GROUP BY project_id, validated_by
329 """
330
331 # Subquery for mapped tasks
332 query_mapped = """
333 SELECT project_id, COUNT(mapped_by) AS mapped
334 FROM tasks
335 WHERE project_id IN (
336 SELECT unnest(projects_mapped) FROM users WHERE id = :user_id
337 ) AND mapped_by = :user_id
338 GROUP BY project_id, mapped_by
339 """
340
341 # Union of validated and mapped tasks
342 query_union = f"""
343 SELECT COALESCE(v.project_id, m.project_id) AS project_id,
344 COALESCE(v.validated, 0) AS validated,
345 COALESCE(m.mapped, 0) AS mapped
346 FROM ({query_validated}) v
347 FULL OUTER JOIN ({query_mapped}) m
348 ON v.project_id = m.project_id
349 """
350
351 # Main query to get project details
352 query_projects = f"""
353 SELECT p.id, p.status, p.default_locale, u.mapped, u.validated, ST_AsGeoJSON(p.centroid) AS centroid
354 FROM projects p
355 JOIN ({query_union}) u ON p.id = u.project_id
356 ORDER BY p.id DESC
357 """
358
359 results = await db.fetch_all(query_projects, {"user_id": user_id})
360
361 mapped_projects_dto = UserMappedProjectsDTO()
362 for row in results:
363 mapped_project = MappedProject()
364 mapped_project.project_id = row["id"]
365 mapped_project.status = ProjectStatus(row["status"]).name
366 mapped_project.tasks_mapped = row["mapped"]
367 mapped_project.tasks_validated = row["validated"]
368 mapped_project.centroid = geojson.loads(row["centroid"])
369 project_info = await ProjectInfo.get_dto_for_locale(
370 db, row["id"], preferred_locale, row["default_locale"]
371 )
372 mapped_project.name = project_info.name
373 mapped_projects_dto.mapped_projects.append(mapped_project)
374 return mapped_projects_dto
