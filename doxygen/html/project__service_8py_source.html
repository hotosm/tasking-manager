126 @staticmethod
127 async def get_contribs_by_day(project_id: int, db: Database) -> ProjectContribsDTO:
128 project = await ProjectService.get_project_by_id(project_id, db)
129 query = """
130 SELECT
131 action_text,
132 DATE(action_date) AS day,
133 task_id
134 FROM task_history
135 WHERE project_id = :project_id
136 AND action = 'STATE_CHANGE'
137 AND action_text IN ('MAPPED', 'VALIDATED', 'INVALIDATED')
138 GROUP BY action_text, day, task_id
139 ORDER BY day ASC
140 """
141 rows = await db.fetch_all(query=query, values={"project_id": project_id})
142
143 contribs_dto = ProjectContribsDTO()
144 dates = sorted({row["day"] for row in rows})
145
146 cumulative_mapped = 0
147 cumulative_validated = 0
148 tasks = {
149 "MAPPED": {"total": 0},
150 "VALIDATED": {"total": 0},
151 "INVALIDATED": {"total": 0},
152 }
153
154 dates_list = []
155 for date in dates:
156 dto = ProjectContribDTO(
157 date=date, mapped=0, validated=0, total_tasks=project.total_tasks
158 )
159
160 values = [
161 (row["action_text"], row["task_id"])
162 for row in rows
163 if row["day"] == date
164 ]
165 values.sort(reverse=True)
166
167 for task_status, task_id in values:
168 if task_status == "MAPPED":
169 if task_id not in tasks["MAPPED"]:
170 tasks["MAPPED"][task_id] = 1
171 tasks["MAPPED"]["total"] += 1
172 dto.mapped += 1
173 elif task_status == "VALIDATED":
174 if task_id not in tasks["VALIDATED"]:
175 tasks["VALIDATED"][task_id] = 1
176 tasks["VALIDATED"]["total"] += 1
177 dto.validated += 1
178 if task_id in tasks["INVALIDATED"]:
179 del tasks["INVALIDATED"][task_id]
180 tasks["INVALIDATED"]["total"] -= 1
181 if task_id not in tasks["MAPPED"]:
182 tasks["MAPPED"][task_id] = 1
183 tasks["MAPPED"]["total"] += 1
184 dto.mapped += 1
185 else: # "INVALIDATED"
186 if task_id not in tasks["INVALIDATED"]:
187 tasks["INVALIDATED"][task_id] = 1
188 tasks["INVALIDATED"]["total"] += 1
189 if task_id in tasks["MAPPED"]:
190 del tasks["MAPPED"][task_id]
191 tasks["MAPPED"]["total"] -= 1
192 if dto.mapped > 0:
193 dto.mapped -= 1
194 if task_id in tasks["VALIDATED"]:
195 del tasks["VALIDATED"][task_id]
196 tasks["VALIDATED"]["total"] -= 1
197 if dto.validated > 0:
198 dto.validated -= 1
199
200 cumulative_mapped = tasks["MAPPED"]["total"]
201 cumulative_validated = tasks["VALIDATED"]["total"]
202 dto.cumulative_mapped = cumulative_mapped
203 dto.cumulative_validated = cumulative_validated
204
205 dates_list.append(dto)
206
207 contribs_dto.stats = dates_list
208 return contribs_dto
