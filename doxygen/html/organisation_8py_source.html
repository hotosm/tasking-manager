65 async def create_from_dto(new_organisation_dto: NewOrganisationDTO, db: Database):
66 """Creates a new organisation from a DTO and associates managers"""
67 slug = new_organisation_dto.slug or slugify(new_organisation_dto.name)
68 query = """
69 INSERT INTO organisations (name, slug, logo, description, url, type, subscription_tier)
70 VALUES (:name, :slug, :logo, :description, :url, :type, :subscription_tier)
71 RETURNING id
72 """
73 values = {
74 "name": new_organisation_dto.name,
75 "slug": slug,
76 "logo": new_organisation_dto.logo,
77 "description": new_organisation_dto.description,
78 "url": new_organisation_dto.url,
79 "type": OrganisationType[new_organisation_dto.type].value,
80 "subscription_tier": new_organisation_dto.subscription_tier,
81 }
82
83 try:
84 organisation_id = await db.execute(query, values)
85
86 for manager in new_organisation_dto.managers:
87 user_query = "SELECT id FROM users WHERE username = :username"
88 user = await db.fetch_one(user_query, {"username": manager})
89
90 if not user:
91 raise NotFound(sub_code="USER_NOT_FOUND", username=manager)
92
93 manager_query = """
94 INSERT INTO organisation_managers (organisation_id, user_id)
95 VALUES (:organisation_id, :user_id)
96 """
97 await db.execute(
98 manager_query,
99 {"organisation_id": organisation_id, "user_id": user.id},
100 )
101
102 return organisation_id
103
104 except Exception as e:
105 raise HTTPException(status_code=500, detail=str(e)) from e

107 async def update(organisation_dto: UpdateOrganisationDTO, db: Database):
108 """Updates Organisation from DTO"""
109 try:
110 org_id = organisation_dto.organisation_id
111 org_dict = organisation_dto.dict(exclude_unset=True)
112 if "type" in org_dict and org_dict["type"] is not None:
113 org_dict["type"] = OrganisationType[org_dict["type"].upper()].value
114
115 update_keys = {
116 key: org_dict[key]
117 for key in org_dict.keys()
118 if key not in ["organisation_id", "managers"]
119 }
120 set_clause = ", ".join(f"{key} = :{key}" for key in update_keys.keys())
121 if set_clause:
122 update_query = f"""
123 UPDATE organisations
124 SET {set_clause}
125 WHERE id = :id
126 """
127 await db.execute(update_query, values={**update_keys, "id": org_id})
128
129 if organisation_dto.managers:
130 clear_managers_query = """
131 DELETE FROM organisation_managers
132 WHERE organisation_id = :id
133 """
134 await db.execute(clear_managers_query, values={"id": org_id})
135 for manager_username in organisation_dto.managers:
136 user_query = "SELECT id FROM users WHERE username = :username"
137 user = await db.fetch_one(
138 user_query, {"username": manager_username}
139 )
140
141 if not user:
142 raise NotFound(
143 sub_code="USER_NOT_FOUND", username=manager_username
144 )
145
146 insert_manager_query = """
147 INSERT INTO organisation_managers (organisation_id, user_id)
148 VALUES (:organisation_id, :user_id)
149 """
150 await db.execute(
151 insert_manager_query,
152 {"organisation_id": org_id, "user_id": user.id},
153 )
154 except Exception as e:

206 async def get_all_organisations(db: Database):
207 """Gets all organisations"""
208 query = """
209 SELECT
210 o.id AS organisation_id,
211 o.name,
212 o.slug,
213 o.logo,
214 o.description,
215 o.url,
216 CASE
217 WHEN o.type = 1 THEN 'FREE'
218 WHEN o.type = 2 THEN 'DISCOUNTED'
219 WHEN o.type = 3 THEN 'FULL_FEE'
220 ELSE 'UNKNOWN'
221 END AS type,
222 o.subscription_tier,
223 COALESCE(
224 json_agg(
225 json_build_object(
226 'id', u.id,
227 'username', u.username,
228 'picture_url', u.picture_url
229 )
230 ) FILTER (WHERE u.id IS NOT NULL), '[]'
231 ) AS managers
232 FROM organisations o
233 LEFT JOIN organisation_managers om ON o.id = om.organisation_id
234 LEFT JOIN users u ON om.user_id = u.id
235 GROUP BY o.id
236 ORDER BY o.name
237 """
238 result = await db.fetch_all(query)
239 return result

267 ) AS managers

273 ORDER BY o.name
