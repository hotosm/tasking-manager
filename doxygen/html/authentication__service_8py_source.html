200 @staticmethod
201 def generate_random_state(length=48, chars=UNICODE_ASCII_CHARACTER_SET):
202 """Generates a non-guessable OAuth token
203 OAuth (1 and 2) does not specify the format of tokens except that they
204 should be strings of random characters. Tokens should not be guessable
205 and entropy when generating the random characters is important. Which is
206 why SystemRandom is used instead of the default random.choice method.
207 """
208 rand = SystemRandom()
209 return "".join(rand.choice(chars) for x in range(length))
