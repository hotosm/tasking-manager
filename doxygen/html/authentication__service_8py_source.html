202 @staticmethod
203 def generate_random_state(length=48, chars=UNICODE_ASCII_CHARACTER_SET):
204 """Generates a non-guessable OAuth token
205 OAuth (1 and 2) does not specify the format of tokens except that they
206 should be strings of random characters. Tokens should not be guessable
207 and entropy when generating the random characters is important. Which is
208 why SystemRandom is used instead of the default random.choice method.
209 """
210 rand = SystemRandom()
211 return "".join(rand.choice(chars) for x in range(length))
