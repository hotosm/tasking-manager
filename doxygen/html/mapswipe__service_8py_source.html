60 async def _post_graphql(self, payload: dict) -> dict:
61 """
62 Async POST GraphQL payload, handle CSRF (Referer/Origin + X-CSRFToken), retry once on 403,
63 check HTTP and GraphQL errors, and return parsed JSON dict.
64 """
65 client = await self._ensure_client()
66 origin = self._get_origin()
67
68 # try to obtain CSRF cookie
69 csrf_token = await self._fetch_csrf_cookie(client)
70
71 headers = {
72 "Content-Type": "application/json",
73 "Accept": "application/json",
74 "Referer": origin,
75 "Origin": origin,
76 }
77 if csrf_token:
78 headers["X-CSRFToken"] = csrf_token
79
80 try:
81 resp = await client.post(MAPSWIPE_API_URL, headers=headers, json=payload)
82 except httpx.RequestError as exc:
83 logger.exception("MapSwipe POST failed: %s", exc)
84 raise Conflict(
85 "MAPSWIPE_NETWORK_ERROR", f"Failed to reach MapSwipe API: {exc}"
86 )
87
88 # retry on 403 after refreshing cookies once
89 if resp.status_code == 403:
90 logger.warning(
91 "MapSwipe responded 403; refreshing cookies and retrying once."
92 )
93 await self._fetch_csrf_cookie(client)
94 # update csrf_token if present
95 for ck in ("MAPSWIPE-PROD-CSRFTOKEN", "csrftoken", "CSRF-TOKEN"):
96 csrf_token = client.cookies.get(ck)
97 if csrf_token:
98 headers["X-CSRFToken"] = csrf_token
99 break
100 else:
101 headers.pop("X-CSRFToken", None)
102
103 try:
104 resp = await client.post(
105 MAPSWIPE_API_URL, headers=headers, json=payload
106 )
107 except httpx.RequestError as exc:
108 logger.exception("MapSwipe POST retry failed: %s", exc)
109 raise Conflict(
110 "MAPSWIPE_NETWORK_ERROR",
111 f"Failed to reach MapSwipe API on retry: {exc}",
112 )
113
114 # HTTP errors
115 if resp.status_code >= 400:
116 logger.error(
117 "MapSwipe HTTP error %s: %s", resp.status_code, resp.text[:2000]
118 )
119 raise Conflict("MAPSWIPE_HTTP_ERROR", f"MapSwipe HTTP {resp.status_code}")
120
121 # parse JSON
122 try:
123 parsed = resp.json()
124 except ValueError:
125 logger.error("MapSwipe returned non-JSON: %s", resp.text[:2000])
126 raise Conflict(
127 "MAPSWIPE_INVALID_RESPONSE", "MapSwipe returned non-JSON response"
128 )
129
130 # GraphQL-level errors
131 if parsed.get("errors"):
132 err_msg = parsed["errors"][0].get("message", "GraphQL error")
133 logger.error(
134 "MapSwipe GraphQL error: %s -- full: %s", err_msg, parsed["errors"]
135 )
136 raise Conflict(
137 "MAPSWIPE_GRAPHQL_ERROR", f"MapSwipe GraphQL error: {err_msg}"
138 )
139
140 # guard against null data
141 if parsed.get("data") is None:
142 logger.error("MapSwipe returned null data: %s", parsed)
143 raise Conflict("MAPSWIPE_NO_DATA", "MapSwipe returned null data")
144
145 return parsed
