100 async def get_dto_for_locale(
101 db: Database, project_id: int, locale: str, default_locale: str = "en"
102 ) -> ProjectInfoDTO:
103 """
104 Gets the ProjectInfoDTO for the project for the requested locale.
105 If not found, then the default locale is used.
106 :param db: The async database connection
107 :param project_id: ProjectID in scope
108 :param locale: Locale requested by user
109 :param default_locale: Default locale of project
110 :return: ProjectInfoDTO
111 :raises: ValueError if no info found for Default Locale
112 """
113 query = """
114 SELECT * FROM project_info
115 WHERE project_id = :project_id AND locale = :locale
116 """
117 # Execute the query for the requested locale
118 project_info = await db.fetch_one(
119 query, values={"project_id": project_id, "locale": locale}
120 )
121 if project_info is None:
122 # Define the SQL query to get project info by default locale
123 query_default = """
124 SELECT * FROM project_info
125 WHERE project_id = :project_id AND locale = :default_locale
126 """
127
128 # Execute the query for the default locale
129 project_info = await db.fetch_one(
130 query_default,
131 values={"project_id": project_id, "default_locale": default_locale},
132 )
133
134 if project_info is None:
135 error_message = f"BAD DATA: No info for project {project_id},locale: {locale},default {default_locale}"
136 raise ValueError(error_message)
137
138 return ProjectInfoDTO(**project_info)
139
140 if locale == default_locale:
141 # Return the DTO for the default locale
142 return ProjectInfoDTO(**project_info)
143 # Define the SQL query to get project info by default locale for partial translations
144 query_default = """
145 SELECT * FROM project_info
146 WHERE project_id = :project_id AND locale = :default_locale
147 """
148
149 # Execute the query for the default locale
150 default_locale_info = await db.fetch_one(
151 query_default,
152 values={"project_id": project_id, "default_locale": default_locale},
153 )
154
155 if default_locale_info is None:
156 error_message = f"BAD DATA: no info for project {project_id}, locale: {locale}, default {default_locale}"
157 raise ValueError(error_message)
158
159 combined_info = ProjectInfoDTO(locale=project_info.locale)
160 combined_info.name = (
161 project_info.name if project_info.name else default_locale_info.name
162 )
163 combined_info.description = (
164 project_info.description
165 if project_info.description
166 else default_locale_info.description
167 )
168 combined_info.short_description = (
169 project_info.short_description
170 if project_info.short_description
171 else default_locale_info.short_description
172 )
173 combined_info.instructions = (
174 project_info.instructions
175 if project_info.instructions
176 else default_locale_info.instructions
177 )
178 combined_info.per_task_instructions = (
179 project_info.per_task_instructions
180 if project_info.per_task_instructions
181 else default_locale_info.per_task_instructions
182 )
