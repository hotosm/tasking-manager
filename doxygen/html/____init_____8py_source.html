65def initialise_logger(app):
66 """
67 Read environment config then initialise a 2MB rotating log. Prod Log Level can be reduced to help diagnose Prod
68 only issues.
69 """
70 log_dir = app.config["LOG_DIR"]
71 log_level = app.config["LOG_LEVEL"]
72 if not os.path.exists(log_dir):
73 os.makedirs(log_dir)
74
75 file_handler = RotatingFileHandler(
76 log_dir + "/tasking-manager.log", "a", 2 * 1024 * 1024, 3
77 )
78 file_handler.setLevel(log_level)
79 file_handler.setFormatter(
80 logging.Formatter(
81 "%(asctime)s %(levelname)s: %(message)s [in %(pathname)s:%(lineno)d]"
82 )
83 )
84 app.logger.addHandler(file_handler)
85 app.logger.setLevel(log_level)
