290 @staticmethod
291 async def generate_gpx(
292 project_id: int, task_ids_str: str, db: Database, timestamp=None
293 ):
294 """
295 Creates a GPX file for supplied tasks. Timestamp is for unit testing only.
296 You can use the following URL to test locally:
297 http://www.openstreetmap.org/edit?editor=id&#map=11/31.50362930069913/34.628906243797054&comment=CHANGSET_COMMENT&gpx=http://localhost:5000/api/v2/projects/{project_id}/tasks/queries/gpx%3Ftasks=2
298 """
299
300 if timestamp is None:
301 timestamp = datetime.datetime.utcnow()
302
303 root = ET.Element(
304 "gpx",
305 attrib=dict(
306 version="1.1",
307 creator="HOT Tasking Manager",
308 xmlns="http://www.topografix.com/GPX/1/1",
309 ),
310 )
311
312 # Create GPX Metadata element
313 metadata = ET.Element("metadata")
314 link = ET.SubElement(
315 metadata,
316 "link",
317 attrib=dict(href="https://github.com/hotosm/tasking-manager"),
318 )
319 ET.SubElement(link, "text").text = "HOT Tasking Manager"
320 ET.SubElement(metadata, "time").text = timestamp.isoformat()
321 root.append(metadata)
322
323 # Create trk element
324 trk = ET.Element("trk")
325 root.append(trk)
326 ET.SubElement(trk, "name").text = (
327 f"Task for project {project_id}. Do not edit outside of this area!"
328 )
329
330 # Construct trkseg elements
331 if task_ids_str is not None:
332 task_ids = list(map(int, task_ids_str.split(",")))
333 tasks = await Task.get_tasks(project_id, task_ids, db)
334 if not tasks or len(tasks) == 0:
335 raise NotFound(
336 sub_code="TASKS_NOT_FOUND", project_id=project_id, task_ids=task_ids
337 )
338 else:
339 tasks = await Task.get_all_tasks(project_id, db)
340 if not tasks or len(tasks) == 0:
341 raise NotFound(sub_code="TASKS_NOT_FOUND", project_id=project_id)
342
343 for task in tasks:
344 # task_geom = shape.to_shape(task.geometry)
345 if isinstance(task["geometry"], (bytes, str)):
346 task_geom = to_shape(WKBElement(task["geometry"], srid=4326))
347 else:
348 raise ValueError("Invalid geometry format")
349 for poly in task_geom.geoms:
350 trkseg = ET.SubElement(trk, "trkseg")
351 for point in poly.exterior.coords:
352 ET.SubElement(
353 trkseg,
354 "trkpt",
355 attrib=dict(lon=str(point[0]), lat=str(point[1])),
356 )
357 wpt = ET.Element(
358 "wpt", attrib=dict(lon=str(point[0]), lat=str(point[1]))
359 )
360 root.append(wpt)
361
362 xml_gpx = ET.tostring(root, encoding="utf8")
363 return xml_gpx

404 visible="true",
405 id=str(fake_id),
406 lon=str(point[0]),
407 lat=str(point[1]),
408 ),
409 )
410 ET.SubElement(way, "nd", attrib=dict(ref=str(fake_id)))
