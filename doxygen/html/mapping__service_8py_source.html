290 async def generate_gpx(
291 project_id: int, task_ids_str: str, db: Database, timestamp=None
292 ):
293 """
294 Creates a GPX file for supplied tasks. Timestamp is for unit testing only.
295 You can use the following URL to test locally:
296 http://www.openstreetmap.org/edit?editor=id&#map=11/31.50362930069913/34.628906243797054&comment=CHANGSET_COMMENT&gpx=http://localhost:5000/api/v2/projects/{project_id}/tasks/queries/gpx%3Ftasks=2
297 """
298
299 if timestamp is None:
300 timestamp = datetime.datetime.utcnow()
301
302 root = ET.Element(
303 "gpx",
304 attrib=dict(
305 version="1.1",
306 creator="HOT Tasking Manager",
307 xmlns="http://www.topografix.com/GPX/1/1",
308 ),
309 )
310
311 # Create GPX Metadata element
312 metadata = ET.Element("metadata")
313 link = ET.SubElement(
314 metadata,
315 "link",
316 attrib=dict(href="https://github.com/hotosm/tasking-manager"),
317 )
318 ET.SubElement(link, "text").text = "HOT Tasking Manager"
319 ET.SubElement(metadata, "time").text = timestamp.isoformat()
320 root.append(metadata)
321
322 # Create trk element
323 trk = ET.Element("trk")
324 root.append(trk)
325 ET.SubElement(trk, "name").text = (
326 f"Task for project {project_id}. Do not edit outside of this area!"
327 )
328
329 # Construct trkseg elements
330 if task_ids_str is not None:
331 task_ids = list(map(int, task_ids_str.split(",")))
332 tasks = await Task.get_tasks(project_id, task_ids, db)
333 if not tasks or len(tasks) == 0:
334 raise NotFound(
335 sub_code="TASKS_NOT_FOUND", project_id=project_id, task_ids=task_ids
336 )
337 else:
338 tasks = await Task.get_all_tasks(project_id, db)
339 if not tasks or len(tasks) == 0:
340 raise NotFound(sub_code="TASKS_NOT_FOUND", project_id=project_id)
341
342 for task in tasks:
343 # task_geom = shape.to_shape(task.geometry)
344 if isinstance(task["geometry"], (bytes, str)):
345 task_geom = to_shape(WKBElement(task["geometry"], srid=4326))
346 else:
347 raise ValueError("Invalid geometry format")
348 for poly in task_geom.geoms:
349 trkseg = ET.SubElement(trk, "trkseg")
350 for point in poly.exterior.coords:
351 ET.SubElement(
352 trkseg,
353 "trkpt",
354 attrib=dict(lon=str(point[0]), lat=str(point[1])),
355 )
356 wpt = ET.Element(
357 "wpt", attrib=dict(lon=str(point[0]), lat=str(point[1]))
358 )
359 root.append(wpt)
360
361 xml_gpx = ET.tostring(root, encoding="utf8")
362 return xml_gpx

404 id=str(fake_id),
